def mgr_bot():
    print("mgr bot: Hi! Ask me about courses, fees, or admission. Type 'bye' to exit.")
    
    # Store college data in a dictionary
    college_data = {
        "btech": {
            "duration": "4 years",
            "fees": "₹1,40,000 per year",
            "eligibility": "12th with PCM, 60%+"
        },
        "mba": {
            "duration": "2 years", 
            "fees": "₹2,50,000 per year",
            "eligibility": "Graduation, 50%+"
        },
        "bca": {
            "duration": "3 years",
            "fees": "₹80,000 per year", 
            "eligibility": "12th with any stream, 50%+"
        },
        "bsc": {
            "duration": "3 years",
            "fees": "₹60,000 per year",
            "eligibility": "12th with Science, 50%+"
        }
    }
    
    while True:
        user_input = input("You: ").lower().strip()
        
        if user_input == 'bye':
            print("mgrBot: All the best for your admission! Bye.")
            break
            
        elif 'course' in user_input or 'list' in user_input:
            courses = ", ".join(college_data.keys()).upper()
            print(f"mgrBot: We offer: {courses}")
            
        elif 'fee' in user_input:
            found = False
            for course in college_data:
                if course in user_input:
                    info = college_data[course]
                    print(f"mgrBot: {course.upper()} fees = {info['fees']}, Duration = {info['duration']}")
                    found = True
                    break
            if not found:
                print("mgrBot: Which course fees? Try: 'btech fees' or 'mba fees'")
                
        elif 'duration' in user_input or 'years' in user_input:
            found = False
            for course in college_data:
                if course in user_input:
                    print(f"mgrBot: {course.upper()} duration = {college_data[course]['duration']}")
                    found = True
                    break
            if not found:
                print("mgrBot: Mention the course name. Example: 'btech duration'")
                
        elif 'eligibility' in user_input:
            found = False
            for course in college_data:
                if course in user_input:
                    print(f"mgrBot: {course.upper()} eligibility = {college_data[course]['eligibility']}")
                    found = True
                    break
            if not found:
                print("mgrBot: Ask like: 'btech eligibility'")
                
        elif 'hello' in user_input or 'hi' in user_input:
            print("mgrBot: Hello! Ask me about courses, fees, duration, or eligibility.")
            
        else:
            print("mgrBot: Sorry, I didn't get that. Try 'list courses', 'btech fees', or 'mba duration'")

mgr_bot()
